Structure:

```js
import { useState, useEffect } from "react";
import { useReducedMotion } from "motion/react";
import { SCENE_DURATIONS } from "../lib/motionTokens";

// Derive count from array — prevents drift if durations change in motionTokens.js
const SCENE_COUNT = SCENE_DURATIONS.length;

export function useMoldingLoop() {
  const [sceneIndex, setSceneIndex] = useState(0);
  const shouldReduceMotion = useReducedMotion();

  useEffect(() => {
    // Freeze at current index when OS prefers reduced motion.
    // The orchestrator (InjectionMoldingLoop) will render the static
    // fallback — this hook just stops advancing.
    if (shouldReduceMotion) return;

    // Convert seconds to milliseconds for setTimeout
    const duration = SCENE_DURATIONS[sceneIndex] * 1000;
    const id = setTimeout(() => {
      setSceneIndex(i => (i + 1) % SCENE_COUNT);
    }, duration);

    // Cleanup fires when sceneIndex or shouldReduceMotion changes,
    // preventing stacked timers. Mirrors observer.disconnect() in useInView.js.
    return () => clearTimeout(id);
  }, [sceneIndex, shouldReduceMotion]);

  return { sceneIndex, shouldReduceMotion };
}
```

Do NOT use a default export — all project hooks use named exports.
Do NOT add a play/pause API — not in Phase 2 scope.
Do NOT derive SCENE_COUNT inline in the modulo — use the const so the reader sees 7 is intentional.
