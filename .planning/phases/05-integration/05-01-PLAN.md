import { m } from "motion/react";
import { REVEAL, EASE_SPRING_DEFAULT } from "../../lib/motionTokens";

Define variant objects at module level (above the component function):

```js
// Badge pill — simple fade
const badgeVariants = {
  hidden: { opacity: 0 },
  show:   { opacity: 1, transition: REVEAL },
};

// Headline + body text — slide up from 24px below
const slideUpVariants = {
  hidden: { opacity: 0, y: 24 },
  show:   { opacity: 1, y: 0, transition: REVEAL },
};

// CTA row — same slide up, slight extra delay applied via custom prop
const ctaVariants = {
  hidden: { opacity: 0, y: 24 },
  show:   { opacity: 1, y: 0, transition: { ...REVEAL, delay: 0.3 } },
};

// Stats cards — stampIn overshoot via spring
const stampVariants = {
  hidden: { opacity: 0, y: -20, scale: 1.15 },
  show: (i) => ({
    opacity: 1, y: 0, scale: 1,
    transition: { ...EASE_SPRING_DEFAULT, delay: i * 0.12 },
  }),
};
```

Element-by-element conversion (preserve ALL className values, only change the tag
and remove CSS animation classes):

2. `<h1 className="... animate-slide-up ...">` →
   `<m.h1 variants={slideUpVariants} initial="hidden" animate="show" ...>`
   Remove `animate-slide-up`.

3. Body `<p className="... animate-slide-up ..." style={{ animationDelay: "0.1s" }}>` →

   Remove `animate-slide-up` and remove `style={{ animationDelay }}`.

4. Tags `<div className="... animate-fade-in ..." style={{ animationDelay: "0.25s" }}>` →

   Remove `animate-fade-in` and remove `style={{ animationDelay }}`.

5. CTA row `<div className="... animate-slide-up ..." style={{ animationDelay: "0.3s" }}>` →
   `<m.div variants={ctaVariants} initial="hidden" animate="show" ...>`
   Remove `animate-slide-up` and remove `style={{ animationDelay }}`.

   CRITICAL: The Button components inside keep their `id` attributes EXACTLY:
